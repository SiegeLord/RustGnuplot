A simple gnuplot controller.

Example

use gnuplot::{Figure, Caption, Color};

let x = &[0u32, 1, 2];
let y = &[3u32, 4, 5];
let mut fg = Figure::new();
fg.axes2d()
.lines(x.iter(), y.iter(), &[Caption("A line"), Color("black")]);
fg.show();
